    import React, { useState, useEffect } from 'react';
    import './LikeButton.scss';
    import axios from 'axios';

    const LikeButton = () => {
        const [likeCount, setLikeCount] = useState(0);
        const [dislikeCount, setDislikeCount] = useState(0);
        const [liked, setLiked] = useState(false);
        const [disliked, setDisliked] = useState(false);

        useEffect(() => {
            axios.get('https://artsolovev.ru/likeData.json')
                .then((response) => {
                    const { likes, dislikes } = response.data;
                    setLikeCount(likes);
                    setDislikeCount(dislikes);
                })
                .catch((error) => {
                    console.error('–û—à–∏–±–∫–∞ –ø—Ä–∏ –∑–∞–≥—Ä—É–∑–∫–µ –¥–∞–Ω–Ω—ã—Ö:', error);
                });
        }, []);

        const handleLike = () => {
            if (liked) {
                setLikeCount(likeCount - 1);
            } else {
                setLikeCount(likeCount + 1);

                if (disliked) {
                    setDisliked(false);
                    setDislikeCount(dislikeCount - 1);
                }
            }
            setLiked(!liked);
        };

        const handleDislike = () => {
            if (disliked) {
                setDislikeCount(dislikeCount - 1);
            } else {
                setDislikeCount(dislikeCount + 1);

                if (liked) {
                    setLiked(false);
                    setLikeCount(likeCount - 1);
                }
            }
            setDisliked(!disliked);
        };

        useEffect(() => {
            const data = { likes: likeCount, dislikes: dislikeCount };
            axios.post('https://artsolovev.ru/likeData.json', data)
                .catch((error) => {
                    console.error('–û—à–∏–±–∫–∞ –ø—Ä–∏ —Å–æ—Ö—Ä–∞–Ω–µ–Ω–∏–∏ –¥–∞–Ω–Ω—ã—Ö:', error);
                });
        }, [likeCount, dislikeCount]);

        return (
            <div className="like-button-container">
                <button
                    className={`like-button ${liked ? 'liked' : ''}`}
                    onClick={handleLike}
                >
                    –õ–∞–π–∫ {likeCount}
                </button>
                <button
                    className={`like-button ${disliked ? 'disliked' : ''}`}
                    onClick={handleDislike}
                >
                    –î–∏–∑–ª–∞–π–∫ {dislikeCount}
                </button>
            </div>
        );
    };

    export default LikeButton;
